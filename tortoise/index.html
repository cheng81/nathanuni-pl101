// Fractal-tree, McLeopold sample // http://nathansuniversity.com/vanilla/discussion/comment/1495#Comment_1495 define curve(size, angle, scale, count) { if (count > 0) { stroke((size * scale) / 10); forward(size); left(angle); curve(size * scale, angle, scale, count - 1); right(90); curve(size * scale * scale, angle, scale, count - 1); left(90); right(angle); right(180); forward(size); left(180); } } right(180); forward(100); right(180); curve(100, 29, 0.80, 10);
// Sierpinski triangle (http://en.wikipedia.org/wiki/Sierpinski_triangle) // Adapted from http://oather.github.com/nu/pl101/l06-1/index.html define unitTri() { forward(1); left(120); forward(1); left(120); forward(1); } define move(x,y) { penUp(); right(90); forward(x); left(90); forward(y); penDown(); } define tri(n) { push(); n := n - 1; if (n == 0) { unitTri(); } if (n > 0) { scale(1/2); tri(n); move(0, 1); tri(n); left(120); move(0, 1); right(120); tri(n); } pop(); } var numIterations; numIterations := 4; right(90); scale(500); move(0.5, -0.5); tri(numIterations);
